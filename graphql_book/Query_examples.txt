Create an author:

mutation {
  createAuthor(name: "Niklas") {
    id
    name
  }
}

Create a book using the autogenerated authorId "1" from last mutation:

mutation {
  createBook(authorId: 1, title: "Niklas' bog", releaseYear: 2024) {
    id
    title
    releaseYear
    author {
      name
    }
  }
}

Get all books:

query {
  books {
    id
    title
    releaseYear
    author {
      name
    }
  }
}

Get all authors:

query {
  authors {
    id
    name
    books {
      title
    }
  }
}
